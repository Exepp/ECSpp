251 template <typename FnType>
252 inline void EntitySpawner::moveEntityHere(Entity ent, EntityList& entList, EntitySpawner& originSpawner, FnType fn)
253 {
254  static_assert(std::is_invocable_v<FnType, Creator&&>);
255  EPP_ASSERT(entList.isValid(ent) && &originSpawner != this && entList.get(ent).spawnerId == originSpawner.spawnerId);
256 
257  PoolIdx oldIdx = entList.get(ent).poolIdx;
258  PoolIdx newIdx(entityPool.data.size());
259  auto oriPoolsPtr = originSpawner.cPools.begin();
260  auto oriPoolsEnd = originSpawner.cPools.end();
261 
262  // moving/creating/destroying components
263  // this loop takes advantage of the fact, that the pools are sorted by their cIds
264  for (auto& pool : cPools) {
265  while (oriPoolsPtr != oriPoolsEnd && oriPoolsPtr->getCId() < pool.getCId()) // omit (destroy) components that were not specified (in archetype) to be in this spawner
266  (oriPoolsPtr++)->destroy(oldIdx.value);
267  pool.alloc(); // only allocates memory (constructor is not called yet)
268  if (oriPoolsPtr != oriPoolsEnd && oriPoolsPtr->getCId() == pool.getCId()) // this component is in the original spawner, move it
269  pool.construct(newIdx.value, (*oriPoolsPtr)[oldIdx.value]); // component from the original will be destroyed either on the next loop or, for the last one, after the whole loop
270  }
271  while (oriPoolsPtr != oriPoolsEnd) // destroy the rest (if there is any)
272  (oriPoolsPtr++)->destroy(oldIdx.value);
273  originSpawner.removeFromEntityPool(oldIdx, entList); // remove entity
274  entityPool.create(ent); // add entity
275  entList.changeEntity(ent, newIdx, spawnerId);
276  fn(Creator(*this, newIdx, originSpawner.mask)); // originSpawner.mask contains all the components that were moved, no need to delete possible excess
277 }
