286 inline void EntityList::reserve(std::size_t newReserved)
287 {
288  EPP_ASSERT(newReserved > reserved);
289  Cell* newMemory = reinterpret_cast<Cell*>(operator new[](sizeof(Cell) * std::uint64_t(newReserved)));
290  if (data) {
291  std::memcpy(newMemory, data, reserved * sizeof(Cell));
292  operator delete[](data);
293  }
294  freeLeft += newReserved - reserved;
295  data = newMemory;
296 
297  // init new elements
298  for (ListIdx i(reserved); i.value < newReserved - 1; ++i.value)
299  new (data + i.value) Cell(Cell::Free{ ListIdx(i.value + 1), EntVersion(0) });
300  new (data + (newReserved - 1)) Cell(Cell::Free{ ListIdx(freeIndex), EntVersion(0) });
301 
302  freeIndex = ListIdx(reserved);
303  reserved = newReserved;
304 }
