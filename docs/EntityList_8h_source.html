291 inline void EntityList::reserve(std::size_t newReserved)
292 {
293  EPP_ASSERT(newReserved > reserved);
294  Cell* newMemory = reinterpret_cast<Cell*>(operator new[](sizeof(Cell) * std::uintptr_t(newReserved)));
295  if (data) {
296  std::memcpy(newMemory, data, reserved * sizeof(Cell));
297  operator delete[](data);
298  }
299  freeLeft += newReserved - reserved;
300  data = newMemory;
301 
302  // init new elements
303  for (ListIdx i(reserved); i.value < newReserved - 1; ++i.value)
304  new (data + i.value) Cell(Cell::Free{ ListIdx(i.value + 1), EntVersion(0) });
305  new (data + (newReserved - 1)) Cell(Cell::Free{ ListIdx(freeIndex), EntVersion(0) });
306 
307  freeIndex = ListIdx(reserved);
308  reserved = newReserved;
309 }
