97  {
98  Iterator_t it = select.begin(), end = select.end(), tempEnd = it;
99  std::size_t const step = isSub ? 0 : stepFn(select.countEntities());

102  std::unique_lock guard(uptask->mutex); // no need to lock this->mutex - only this thread can modify its fence
103  uptask->condVar.wait(guard, [&] { return fence < uptask->fence; });
104  tempEnd.jumpToOrBeyond(uptask->fence.sId, uptask->fence.pIdx);
