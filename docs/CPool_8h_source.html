243 inline void CPool::reserve(std::size_t newReserved)
244 {
245  if (newReserved == reserved)
246  return;
247  void* newData = newReserved ? operator new[](metadata.size* newReserved, std::align_val_t(metadata.alignment))
248  : nullptr;
249  if (data) {
250  auto toMove = std::min(newReserved, dataUsed);
251  for (Idx_t i = 0; i < toMove; ++i)
252  metadata.moveConstructor(addressAtIdx(newData, i), addressAtIdx(i));
253  for (Idx_t i = 0; i < dataUsed; ++i)
254  metadata.destructor(addressAtIdx(i));
255  dataUsed = toMove;
256  operator delete[](data, std::align_val_t(metadata.alignment));
257  }
258  data = newData;
259  reserved = newReserved;
260 }
